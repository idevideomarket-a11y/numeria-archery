#### Reminder Implementation

```javascript
// REMINDER SYSTEM LOGIC
// =====================

const REMINDER_CONFIG = {
    badgeOnMenu: true,           // Always show until converted
    bannerAfterPlays: 3,         // Show banner after 3 free level plays
    reshowScreenDays: 7,         // Re-show conversion screen after 7 days
    finalOfferDays: 15,          // Final reminder with discount
};

// Check if should show reminder
function shouldShowReminder() {
    const dismissed = localStorage.getItem('conversionDismissed');
    const dismissedAt = localStorage.getItem('dismissedAt');
    
    if (!dismissed) return false;
    
    const daysSince = (Date.now() - parseInt(dismissedAt)) / (1000 * 60 * 60 * 24);
    
    // Re-show after 7 days
    if (daysSince >= 7 && daysSince < 15) {
        return 'reshow';
    }
    
    // Final offer after 15 days
    if (daysSince >= 15) {
        return 'final_offer';
    }
    
    return 'badge_only';
}

// Save dismissal
function onDismissConversion() {
    localStorage.setItem('conversionDismissed', 'true');
    localStorage.setItem('dismissedAt', Date.now().toString());
    localStorage.setItem('freePlaysSinceDismiss', '0');
}
```

```javascript
// PREMIUM PAGE LOGIC
// ==================

// URL: game.com/premium

function initPremiumPage() {
    const urlParams = new URLSearchParams(window.location.search);
    const token = urlParams.get('token');
    
    if (!token) {
        showError("Invalid access link. Please contact @DenisBrandMenedjer");
        return;
    }
    
    // Validate token (in production, verify with server)
    if (isValidToken(token)) {
        // Set premium flag
        localStorage.setItem('numeria_premium', 'true');
        localStorage.setItem('numeria_token', token);
        localStorage.setItem('numeria_unlockedAt', Date.now().toString());
        
        showSuccess("Premium unlocked! Click below to play.");
        showOpenGameButton();
    } else {
        showError("This link has expired or is invalid.");
    }
}

// Game checks on load
function checkPremiumStatus() {
    const isPremium = localStorage.getItem('numeria_premium') === 'true';
    const token = localStorage.getItem('numeria_token');
    
    return {
        isPremium: isPremium && token !== null,
        maxLevel: isPremium ? 50 : 10,
        unlockedWorlds: isPremium ? 6 : 1
    };
}

// Token validation (server-side in production)
function isValidToken(token) {
    // Check format: 8 alphanumeric characters
    const tokenRegex = /^[A-Z0-9]{8}$/;
    return tokenRegex.test(token);
}
```

#### Token Generation (Founder Side)

```javascript
// TOKEN GENERATION FOR FOUNDER
// ============================

function generatePremiumToken() {
    const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789'; // No confusing chars
    let token = '';
    
    for (let i = 0; i < 8; i++) {
        token += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    
    // Store in database with timestamp
    saveTokenToDatabase({
        token: token,
        createdAt: Date.now(),
        expiresAt: Date.now() + (30 * 24 * 60 * 60 * 1000), // 30 days
        used: false,
        usedAt: null,
        usedBy: null
    });
    
    return token;
}

// Generate link for Telegram
function generatePremiumLink() {
    const token = generatePremiumToken();
    return `https://game.com/premium?token=${token}`;
}
```

```javascript
// MUSIC STATE MANAGEMENT
// ======================

const MusicManager = {
    currentTrack: null,
    volume: 0.4,
    isMuted: false,
    
    tracks: {
        menu: 'bgm_menu.mp3',
        gameplay_forest: 'bgm_gameplay_forest.mp3',
        gameplay_premium: 'bgm_gameplay_premium.mp3'
    },
    
    // Play music based on game state
    playForState(state) {
        if (this.isMuted) return;
        
        let track = null;
        
        switch(state) {
            case 'menu':
            case 'world_select':
            case 'settings':
                track = this.tracks.menu;
                break;
            case 'gameplay':
                const world = GameState.currentWorld;
                if (world === 'forest') {
                    track = this.tracks.gameplay_forest;
                } else {
                    track = this.tracks.gameplay_premium;
                }
                break;
            case 'level_complete':
            case 'conversion_screen':
                // Fade out music, let SFX take over
                this.fadeOut(1000);
                return;
        }
        
        if (track && track !== this.currentTrack) {
            this.play(track);
        }
    },
    
    // Smooth transitions
    fadeOut(duration = 1000) {
        // Gradually reduce volume to 0
        // Then stop playback
    },
    
    fadeIn(track, duration = 1000) {
        // Start at 0 volume
        // Gradually increase to target volume
    }
};
```

#### Mute Toggle Implementation

```javascript
// AUDIO SETTINGS MANAGER
// ======================

const AudioSettings = {
    // Default settings
    defaults: {
        musicEnabled: true,
        musicVolume: 0.4,
        sfxEnabled: true,
        sfxVolume: 0.7,
        masterMuted: false
    },
    
    // Current settings
    settings: {},
    
    // Initialize - load from localStorage
    init() {
        const saved = localStorage.getItem('numeria_audio_settings');
        this.settings = saved ? JSON.parse(saved) : { ...this.defaults };
        this.applySettings();
    },
    
    // Save to localStorage
    save() {
        localStorage.setItem('numeria_audio_settings', JSON.stringify(this.settings));
        this.applySettings();
    },
    
    // Apply current settings
    applySettings() {
        const { musicEnabled, musicVolume, sfxEnabled, sfxVolume, masterMuted } = this.settings;
        
        // Master mute overrides everything
        if (masterMuted) {
            MusicManager.mute();
            SFXManager.mute();
        } else {
            MusicManager.setVolume(musicEnabled ? musicVolume : 0);
            SFXManager.setVolume(sfxEnabled ? sfxVolume : 0);
        }
    },
    
    // Toggle methods
    toggleMusic() {
        this.settings.musicEnabled = !this.settings.musicEnabled;
        this.save();
    },
    
    toggleSFX() {
        this.settings.sfxEnabled = !this.settings.sfxEnabled;
        this.save();
    },
    
    toggleMasterMute() {
        this.settings.masterMuted = !this.settings.masterMuted;
        this.save();
    },
    
    // Volume setters
    setMusicVolume(volume) {
        this.settings.musicVolume = Math.max(0, Math.min(1, volume));
        this.save();
    },
    
    setSFXVolume(volume) {
        this.settings.sfxVolume = Math.max(0, Math.min(1, volume));
        this.save();
    }
};

// Initialize on game load
AudioSettings.init();
```

### 13.8 RECOMMENDED LIGHTWEIGHT IMPLEMENTATION

#### Audio Engine Architecture

```javascript
// LIGHTWEIGHT AUDIO ENGINE
// ========================

class AudioEngine {
    constructor() {
        this.context = null;
        this.sounds = new Map();
        this.music = null;
        this.settings = {
            musicVolume: 0.4,
            sfxVolume: 0.7,
            masterMuted: false
        };
    }
    
    // Initialize audio context (user interaction required)
    init() {
        if (!this.context) {
            this.context = new (window.AudioContext || window.webkitAudioContext)();
        }
        if (this.context.state === 'suspended') {
            this.context.resume();
        }
    }
    
    // Load sound file
    async loadSound(name, url) {
        try {
            const response = await fetch(url);
            const arrayBuffer = await response.arrayBuffer();
            const audioBuffer = await this.context.decodeAudioData(arrayBuffer);
            this.sounds.set(name, audioBuffer);
        } catch (error) {
            console.warn(`Failed to load sound: ${name}`, error);
        }
    }
    
    // Play sound effect
    playSFX(name, options = {}) {
        if (this.settings.masterMuted) return;
        
        const buffer = this.sounds.get(name);
        if (!buffer) return;
        
        const source = this.context.createBufferSource();
        const gainNode = this.context.createGain();
        
        source.buffer = buffer;
        source.playbackRate.value = options.pitch || 1.0;
        
        gainNode.gain.value = this.settings.sfxVolume;
        
        source.connect(gainNode);
        gainNode.connect(this.context.destination);
        
        source.start(0);
    }
    
    // Play background music
    playMusic(name) {
        if (this.settings.masterMuted || !this.settings.musicVolume) return;
        
        // Stop current music
        if (this.music) {
            this.music.stop();
        }
        
        const buffer = this.sounds.get(name);
        if (!buffer) return;
        
        const source = this.context.createBufferSource();
        const gainNode = this.context.createGain();
        
        source.buffer = buffer;
        source.loop = true;
        
        gainNode.gain.value = this.settings.musicVolume;
        
        source.connect(gainNode);
        gainNode.connect(this.context.destination);
        
        source.start(0);
        this.music = source;
    }
    
    // Stop music
    stopMusic() {
        if (this.music) {
            this.music.stop();
            this.music = null;
        }
    }
    
    // Set volumes
    setMusicVolume(volume) {
        this.settings.musicVolume = volume;
    }
    
    setSFXVolume(volume) {
        this.settings.sfxVolume = volume;
    }
    
    // Mute/Unmute
    setMuted(muted) {
        this.settings.masterMuted = muted;
        if (muted) {
            this.stopMusic();
        }
    }
}

// Global instance
const audioEngine = new AudioEngine();

// Initialize on first user interaction
document.addEventListener('click', () => {
    audioEngine.init();
}, { once: true });
```

### 13.10 FORMAT RECOMMENDATIONS

#### Audio Format Guidelines

```
┌─────────────────────────────────────────────────────────────────┐
│              AUDIO FORMAT RECOMMENDATIONS                       │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  FORMAT: MP3 (Primary)                                          │
│  ─────────────────────                                          │
│  • Universal browser support                                    │
│  • Good compression (10:1 ratio)                                │
│  • Small file sizes for web                                     │
│  • Recommended: 128kbps bitrate                                 │
│  • Mono for SFX (smaller files)                                 │
│  • Stereo for music                                             │
│                                                                 │
│  FORMAT: WAV (Fallback)                                         │
│  ──────────────────────                                         │
│  • Uncompressed, highest quality                                │
│  • Use for: Short SFX if MP3 has artifacts                      │
│  • Larger files - use sparingly                                 │
│  • 16-bit, 44.1kHz standard                                     │
│                                                                 │
│  FORMAT: OGG (Optional)                                         │
│  ──────────────────────                                         │
│  • Better compression than MP3                                  │
│  • Open source, no licensing                                    │
│  • Not supported by Safari (iOS)                                │
│  • Use as fallback only                                         │
│                                                                 │
│  RECOMMENDED APPROACH:                                          │
│  ─────────────────────                                          │
│  • All files: MP3 128kbps                                       │
│  • Critical SFX: WAV backup                                     │
│  • No OGG needed for MVP                                        │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```
